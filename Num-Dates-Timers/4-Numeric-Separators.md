### Numeric Separators

```javascript
// Numeric Separators
// 287,460,000,000
const diameter = 287_460_000_000;
console.log(diameter);
const price = 345_99;
console.log(price);
const transferFee1 = 15_00;
const transferFee2 = 1_500;
const PI = 3.1415;
console.log(PI);
console.log(Number("230_000"));
console.log(parseInt("230_000"));
```
